current_track (name=nil,group=nil)

If name is specified, this will associate the current thread with that Track name (the Track will be created if necessary).

If track is nil, it returns the Track object for the Track associated to the current thread.

[show source]

63:   def Benelux.current_track(name=nil,group=nil)
64:     if name.nil?
65:       name = Thread.current.track_name
66:     else
67:       Thread.current.track_name = name
68:       @@mutex.synchronize do
69:         Thread.current.timeline ||= Benelux::Timeline.new
70:         Thread.current.rotated_timelines ||= []
71:         @tracks[name] ||= Track.new(name, group)
72:         @tracks[name].add_thread Thread.current
73:         @known_threads << Thread.current
74:       end
75:     end
76:     Benelux.track(name)
77:   end

update_global_timeline ()

Only updates data from threads that are dead and rotated timelines.

[show source] 
     # File lib/benelux.rb, line 82
 82:   def Benelux.update_global_timeline
 83:     @@mutex.synchronize do
 84:       dthreads = Benelux.known_threads.select { |t| 
 85:         !t.timeline.nil? && (t.nil? || !t.status) &&
 86:         !@processed_dead_threads.member?(t)
 87:       }
 88:       # Threads that have rotated timelines
 89:       rthreads = Benelux.known_threads.select { |t|
 90:         !t.rotated_timelines.empty?
 91:       }
 92:       dtimelines = dthreads.collect { |t| t.timeline }
 93:       # Also get all rotated timelines. 
 94:       rthreads.each { |t| 
 95:         # We loop carefully through the rotated timelines
 96:         # incase something was added while we're working. 
 97:         while !t.rotated_timelines.empty?
 98:           dtimelines.push t.rotated_timelines.shift 
 99:         end
100:       }
101:       Benelux.ld [:update_global_timeline, dthreads.size, rthreads.size, dtimelines.size].inspect
102:       tl = Benelux.timeline.merge! *dtimelines
103:       @processed_dead_threads.push *dthreads
104:       tl
105:     end
106:   end
